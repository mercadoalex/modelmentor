      const { data, error } = await supabase

        .select('*')
        .eq('user_id', userId)
        .order('created_at', { ascending: false });

      if (error) {

        return this.getDefaultStudentData();

      // Aggregate quiz data

          quiz.questions.forEach((q: any) => {
